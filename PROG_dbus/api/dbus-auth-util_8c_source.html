00103

00105

00107         {
00108           _dbus_string_free (&full_path);

00112         _dbus_string_free (&full_path);
00113     }
00114

00117       _dbus_warn ("Could not get next file in %s: %s\n",
00118                   _dbus_string_get_const_data (&test_directory), error.message);
