00077       _dbus_win_free_error_string (emsg);
00078       return -1;
00079     }
00080 
00081   if (bytes_read)
00082     {
00083       /* put length back (doesn't actually realloc) */
00084       _dbus_string_set_length (buffer, start + bytes_read);
00085
